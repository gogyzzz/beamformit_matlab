function [max_val, idx] = maxk(list, k, nmask)
for i = 2:(length(list(:))-1)
if list(i-1) < list(i) && list(i+1) < list(i)
list(i-1) = -9999;
list(i+1) = -9999;
end
end
max_val = zeros(k, 1);
idx = zeros(k,1);
for i = 1:k
[max_val(i), idx(i)] = max(list);
st = max(idx-nmask+1, 1);
ed = min(length(list(:)), idx+nmask-1);
list(st:ed) = 0;
end
